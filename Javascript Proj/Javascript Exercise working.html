Sum and Multiply Numbers

This problem accepts multiple numbers, separated by a comma and returns the sum of the numbers added AND the sum of multiplying each of the numbers and adding the results together



                                            function sumMult(p2Arr) {
                                                var lgArr = [];
                                                lgArr = p2Arr.split(",");
                                                sumTot = sum(lgArr);
                                                multTot = mult(lgArr);
                                                document.getElementById('prob2Output').value = sumTot + "," + multTot;
                                            }
                                            function sum(numArray) {
                                                var sumTot = 0;
                                                for (var i = 0; i < numArray.length; i++) {
                                                    sumTot = sumTot + parseInt(numArray[i]);
                                                }
                                                return sumTot;
                                            }
        
                                            function mult(numArray) {
                                                var multTot = 0;
                                                for (var i = 0; i < numArray.length; i++) {
                                                    if (i === 0) {
                                                        multTot = parseInt(numArray[i]);
                                                    }
                                                    else {
                                                        multTot = multTot * parseInt(numArray[i]);
                                                    }
                                                }
                                                return multTot;
                                            }

Find Factorial of a Number

This problem accepts a number as input and calculates the factorial of the number

The factorial of a non-negative integer n denoted by n!, is the product of all positive integers less than or equal to n.

(i.e. 5!=5x4x3x2x1 = 120)



                                            function fact(numFact) {
                                                var prodTot = numFact;
                                                for (i = numFact-1; i > 1; i--) {
                                                    prodTot = prodTot * i;
                                                }
                                                document.getElementById('prob3Output').value = prodTot;
                                            }

A Perfect Number is a positive integer that is equal to the sum of its proper divisors excluding the number itself

(i.e. 6 is an Perfect Number since 1 + 2 + 3 = 6)

Happy Numbers

This problem will calculate and display the first 5 Happy Numbers

Happy Numbers are positive integers that can be reduced to one (1) by replacing the number with the sum of the square of its digits and repeat until the result is 1 or an endless loop. The endless loop is prevented by tracking the numbers calculated and when a repeat occurs, the loop ends.


                                            function fndHappy() {
                                                var cnt = 1;
                                                var number = 1;
                                                var arrHappy = [];
                                                var n = 0;
                                                var chk = [];
                                                arrHappy.push(number);
                                                number++;
                                                // while number is not 1
                                                while (n != 1) {
                                                    var digits = getDigits(n, 2);      // get computed power for each digit
                                                    var n = sum(digits);                    // sum the computed powers
                                                    if (n != 1) {
                                                        var x = chk.indexOf(n)
                                                        if (x < 0){
                                                            chk.push(n)
                                                        }
                                                        else {
                                                            chk = [];
                                                            number++;
                                                            n = number;
                                                        }
                                                    }
                                                    else {
                                                        arrHappy.push(number);
                                                        cnt++;
                                                        number++;
                                                        n = number;
                                                    }
                                                    if (cnt === 5) {
                                                        document.getElementById('prob7Output').value = arrHappy;
                                                        n = 1;
                                                    }
                                                }
                                            }

Armstrong Numbers

This problem will calculate and display all 3 digit Armstrong Numbers

An Armstrong Number is an integer such that the sum of the cube of it's digits is equal to the number itself

(i.e. 371 is an Armstrong number since 3cubed + 7cubed + 1cubed = 371)


                                                                                            
                                            function fndAs() {
                                                var asArr = [];
                                                for (x = 100; x < 1000; x++) {
                                                    var digits = getDigits(x, 3);  // get total of digits to given power
                                                    var digitSum = sum(digits);
                                                    if (x == digitSum) {
                                                        asArr.push(x);
                                                    }
                                                }
                                                document.getElementById('prob8Output').value = asArr;
                                            }
                                            function getDigits(number, power) {
                                                if (typeof (power) == 'undefined')
                                                    power = 1;
        
                                                var n = (number + "").split("");    // make n an array of digits
                                                // loop through n and get the power of each digit
                                                for (i = 0; i < n.length; i++) {
                                                    n[i] = Math.pow(n[i], power);
                                                }
                                                return n;
                                            }
